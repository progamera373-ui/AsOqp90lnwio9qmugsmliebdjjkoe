local VALID_KEY = "Nexxa1"

local function normalizeKey(str)
	return string.gsub(string.upper(str), "[%s%-_]", "")
end

local function setStatus(msg, color)
	status.Text = msg
	status.TextColor3 = color or Color3.fromRGB(148, 155, 164)
end

enterBtn.MouseButton1Click:Connect(function()
	local input = textBox.Text
	if input == "" then
		setStatus("Por favor ingresa una clave", Color3.fromRGB(231, 76, 60))
		return
	end

	if normalizeKey(input) == string.upper(VALID_KEY) then
		setStatus("Sistema activado correctamente", Color3.fromRGB(46, 204, 113))
		task.wait(0.5)
		gui:Destroy()
		showWorldSelection()
	else
		setStatus("incorrecta", Color3.fromRGB(231, 76, 60))
		textBox.Text = ""
		textBox:CaptureFocus()
	end
end)

textBox.FocusLost:Connect(function(enterPressed)
	if enterPressed then
		enterBtn.MouseButton1Click:Fire()
	end
end)

function loadWorld1()
	local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/progamera373-ui/Nexa-Library-main-library-ui/refs/heads/main/NexaHub%20main%20.lua", true))()
	local mainWindow = library:AddWindow("                 NEXA HUB | WORLD 1    " .. player.DisplayName, {
		title_bar = {
			Color3.fromRGB(0, 100, 0),
			Color3.fromRGB(0, 160, 0),
			Color3.fromRGB(46, 204, 113)
		},
		title_bar_transparency = 0.2,
		background = {
			Color3.fromRGB(0, 30, 0),
			Color3.fromRGB(0, 60, 0),
			Color3.fromRGB(0, 90, 0)
		},
		background_transparency = 0.1,
		main_color = Color3.fromRGB(46, 204, 113),
		min_size = Vector2.new(650, 650),
		can_resize = true,
	})

	local separator = string.rep("═", 38)
	local allLabels = {}

	local function addRainbowLabel(tab, text, textSize)
		local lbl = tab:AddLabel(text)
		if textSize then lbl.TextSize = textSize end
		table.insert(allLabels, lbl)
		return lbl
	end

	local infoTab = mainWindow:AddTab("InfoTab")
	addRainbowLabel(infoTab, separator)
	addRainbowLabel(infoTab, "WELCOME TO NEXA HUB  FREMIUN", 28)
	addRainbowLabel(infoTab, separator)
	addRainbowLabel(infoTab, "Made by: ALAN||")
	addRainbowLabel(infoTab, "Version: v1.")
	addRainbowLabel(infoTab, separator)
	addRainbowLabel(infoTab, "My Friends:")
	addRainbowLabel(infoTab, "PANA TQ")
	addRainbowLabel(infoTab, "CRIWS")

	task.spawn(function()
		while true do
			local hue = tick() * 0.2 % 1
			local color = Color3.fromHSV(hue, 0.9, 1)
			for _, lbl in ipairs(allLabels) do
				pcall(function() lbl.TextColor3 = color end)
			end
			task.wait(0.02)
		end
	end)

	local MainTab = mainWindow:AddTab("Main")

	local pathsW1 = loadstring(game:HttpGet("https://raw.githubusercontent.com/progamera373-ui/Local-parths-v2/refs/heads/main/LocalPlayer.lua"))()
	local selectedPathW1 = nil
	local autoWalkEnabledW1 = false
	local walkSpeedW1 = 100
	local currentTweenW1 = nil

	local EXACT_NAMES_TO_DESTROY_W1 = {
		"CorridorTrap","CorridorTrap2","LavaTower","NPC_Zone10","NPC_Zone12","NPC & Piege","NPC15"
	}

	local function isExactNameTargetedW1(name)
		for _, target in ipairs(EXACT_NAMES_TO_DESTROY_W1) do
			if name == target then return true end
		end
		return false
	end

	local destroyedSetW1 = setmetatable({}, {__mode = "k"})

	local function safeDestroyW1(instance)
		if destroyedSetW1[instance] then return end
		destroyedSetW1[instance] = true
		pcall(function() instance:Destroy() end)
	end

	local function removeObstaclesW1()
		local removedCount = 0
		for _, descendant in ipairs(Workspace:GetDescendants()) do
			if descendant.Parent and isExactNameTargetedW1(descendant.Name) then
				safeDestroyW1(descendant)
				removedCount = removedCount + 1
			end
		end
		print("[NexaHub] REMOVE OBSTACLES W1: " .. removedCount .. " removed.")
	end

	local pathDropdownW1 = MainTab:AddDropdown("Select Distance", function(value)
		selectedPathW1 = pathsW1[value]
	end)
	pathDropdownW1:Add("10k")
	pathDropdownW1:Add("25k")
	pathDropdownW1:Add("50k")
	pathDropdownW1:Add("150k")

	local speedDropdownW1 = MainTab:AddDropdown("Walk Speed", function(value)
		walkSpeedW1 = tonumber(value)
	end)
	speedDropdownW1:Add("50")
	speedDropdownW1:Add("100")
	speedDropdownW1:Add("150")
	speedDropdownW1:Add("200")
	speedDropdownW1:Add("300")

    -- SWITCH AUTO WIM RECUPERADO
	MainTab:AddSwitch("Auto Wim", function(enabled)
		autoWalkEnabledW1 = enabled
		if enabled then
			removeObstaclesW1()
		else
			if currentTweenW1 then currentTweenW1:Cancel(); currentTweenW1 = nil end
		end
	end)

	MainTab:AddButton("REMOVE OBSTACLES", function() removeObstaclesW1() end)

	task.spawn(function()
		while task.wait() do
			if autoWalkEnabledW1 and selectedPathW1 and #selectedPathW1 > 0 then
				local char = player.Character
				if not char or not char:FindFirstChild("HumanoidRootPart") then task.wait(1); continue end
				local hrp = char.HumanoidRootPart
				for _, p in ipairs(char:GetDescendants()) do
					if p:IsA("BasePart") and p.CanCollide then p.CanCollide = false end
				end
				for i = 1, #selectedPathW1 do
					if not autoWalkEnabledW1 then break end
					local pos = selectedPathW1[i]
					while autoWalkEnabledW1 and (hrp.Position - pos).Magnitude > 2 do
						hrp.Velocity = Vector3.zero
						hrp.RotVelocity = Vector3.zero
						local dist = (hrp.Position - pos).Magnitude
						local t = TweenInfo.new(dist / walkSpeedW1, Enum.EasingStyle.Linear)
						currentTweenW1 = game:GetService("TweenService"):Create(hrp, t, {CFrame = CFrame.new(pos)})
						currentTweenW1:Play()
						currentTweenW1.Completed:Wait()
						currentTweenW1 = nil
						task.wait()
					end
				end
				if autoWalkEnabledW1 and char:FindFirstChild("Humanoid") then
					char.Humanoid.Jump = true
					pcall(function() game:GetService("VirtualInputManager"):SendKeyEvent(true, Enum.KeyCode.One, false, game) end)
					task.wait(1)
					pcall(function() game:GetService("VirtualInputManager"):SendKeyEvent(false, Enum.KeyCode.One, false, game) end)
				end
			end
		end
	end)

	local SettingsTab = mainWindow:AddTab("Settings")
	local antiPauseEnabled = false
	SettingsTab:AddSwitch("Anti Gameplay Paused", function(enabled)
		antiPauseEnabled = enabled
		if enabled then
			task.spawn(function()
				while antiPauseEnabled do
					pcall(function() player.GameplayPaused = false end)
					task.wait(0.5)
				end
			end)
		end
	end)
end

	local function isExactNameTargetedW3(name)
		if PROTECTED_NAMES_W3[name] then return false end
		for _, target in ipairs(EXACT_NAMES_TO_DESTROY_W3) do
			if name == target then return true end
		end
		return false
	end

	local destroyedSetW3 = setmetatable({}, {__mode = "k"})

	local function safeDestroyW3(instance)
		if destroyedSetW3[instance] then return end
		destroyedSetW3[instance] = true
		pcall(function() instance:Destroy() end)
	end

	local function removeObstaclesW3()
		local removedCount = 0
		for _, descendant in ipairs(Workspace:GetDescendants()) do
			if descendant.Parent and isExactNameTargetedW3(descendant.Name) then
				safeDestroyW3(descendant)
				removedCount = removedCount + 1
			end
		end
		print("[NexaHub] REMOVE OBSTACLES W3: " .. removedCount .. " removed.")
	end
